So func0 for 151151151 gives 152152152. This seems interesting. It could be that func0(c) just returns the successor of c. Let's prove this by writing this in python.

def func0(a: int) -> int:
    i: int = 1
    while i & a:
        i <<= 1
    
    j: int = i | a
    while True:
        i >>= 1
        if not i:
            break
        j &= ~i
    return j

After observing this function for a few non-negative values of a, it becomes clear that c = func0(c) resolves to c++ as func0(c) = c+1. This was what the challenge description was about.

# [816696039, ..., 573068354]
nums = [int(num) for num in open('output.txt').read().split(', ')]

solutions = []
for r1 in range(256):
    for r2 in range(256):                    
        num = 0 # for the first iteration

        # amateursCTF{...}
        # we know that the flag starts with 'a'
        x = num + ord('a')
        y = x * r1 % p
        num = pow(y, r2, p)
        
        # compare the result with the given output value
        if num == nums[0]:
            solutions.append((r1, r2))

assert len(solutions) == 1

r1, r2 = solutions[0]

assert r1 == 237
assert r2 == 41

num = 0
flag = ''
for c in nums:
    # totient
    phi = p - 1
    
    # RSA
    d = pow(r2, -1, phi)
    y = pow(c, d, p)
    
    # x * r1 = 1 (mod p)
    x = pow(r1, -1, p) * y % p
    # remember to subtract num
    x = x - num
    
    flag += chr(x)
    num = c
    
print(flag, end='')
